-- MySQL Schema

CREATE TABLE members (
    member_id   VARCHAR(20) PRIMARY KEY,
    name        VARCHAR(100) NOT NULL,
    status      ENUM('active', 'inactive') NOT NULL,
    plan_type   VARCHAR(20) NOT NULL DEFAULT 'basic',
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE providers (
    provider_id VARCHAR(20) PRIMARY KEY,
    name        VARCHAR(100) NOT NULL,
    facility    VARCHAR(50) NOT NULL,
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE procedures (
    procedure_code VARCHAR(20) PRIMARY KEY,
    name           VARCHAR(100) NOT NULL,
    average_cost   DECIMAL(12,2) NOT NULL,
    benefit_limit  DECIMAL(12,2) NOT NULL
);

CREATE TABLE claims (
    claim_id        VARCHAR(20) PRIMARY KEY,
    member_id       VARCHAR(20) NOT NULL,
    provider_id     VARCHAR(20) NOT NULL,
    diagnosis_code  VARCHAR(20) NOT NULL,
    procedure_code  VARCHAR(20) NOT NULL,
    claim_amount    DECIMAL(12,2) NOT NULL,
    approved_amount DECIMAL(12,2),
    status          ENUM('APPROVED', 'PARTIAL', 'REJECTED') NOT NULL,
    fraud_flag      BOOLEAN DEFAULT FALSE,
    rejection_reasons JSON,              -- MySQL JSON type instead of array
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (member_id) REFERENCES members(member_id),
    FOREIGN KEY (provider_id) REFERENCES providers(provider_id),
    FOREIGN KEY (procedure_code) REFERENCES procedures(procedure_code)
);

CREATE INDEX idx_claims_member ON claims(member_id);
CREATE INDEX idx_claims_status ON claims(status);
